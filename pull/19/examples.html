List of examples¶

Your first NumPy code running on GPU¶
import dpnp as np

x = np.asarray([1, 2, 3])
y = np.sum(x)

print(y.shape)  # Must be 0-dimensional array
print(y)  # Expect 6
Select device type while creating array¶
import dpnp as np

x = np.empty(3)
try:
    x = np.asarray([1, 2, 3], device="gpu")
except:
    print("GPU device is not available")

y = np.sum(x)
print(y)  # Expect 6
Compile dpnp code with numba-dpex¶
import dpnp as np
from numba_dpex import jit

@njit(parallel=True, fastmath=True)
def sum_it(x):
    return np.sum(x)


x = None
try:
    x = np.asarray([1, 2, 3], device="gpu")
except:
    print("GPU device is not available")

y = sum_it(x)

print(y.shape)  # Must be 0-dimensional array
print(y)  # Expect 6
